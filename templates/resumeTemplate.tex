%----------------------------------------------------------------------------------------
%	PACKAGES AND OTHER DOCUMENT CONFIGURATIONS
%----------------------------------------------------------------------------------------

\documentclass{resume} % Use the custom resume.cls style

\usepackage[left=0.5in,top=0.4in,right=0.5in,bottom=0.4in]{geometry} % Document margins

\name{<$ basics.name $>} % Your name
\title{<$ basics.label $>} % Job Title
\address{<$ basics.location.city $>, <$ basics.location.region $> <$ basics.location.countryCode $>} % Your address
\address{<$ basics.email $>}
\address{<$ basics.website $>}

\begin{document}


<% macro printDate(date) %>
<$ date | DateYearMonth if date else "Present" $>
<% endmacro %>


%----------------------------------------------------------------------------------------
%	WORK EXPERIENCE SECTION
%----------------------------------------------------------------------------------------

\begin{rSection}{Experience}

<% for workplace in work%>

<% set isNotFirst = joiner("true") %>

<% for job in workplace.roles %>

\begin{rSubsection}{<$ workplace.company  if not isNotFirst() $>}{ <$ workplace.title $> }{<$ job.position $>}{<$ printDate(job.startDate) $> --- <$ printDate(job.endDate) $>}
{ <$ job.summary | TeXEscape $> }
<% for highlight in job.highlights %>
\item <$ highlight | TeXEscape $>
<% endfor %>
\end{rSubsection}

<% endfor %>

<% endfor %>

\end{rSection}

%----------------------------------------------------------------------------------------
%	AWARDS SECTION
%----------------------------------------------------------------------------------------

\begin{rSection}{Awards}

<% for award in awards %>

{\bf <$ award.awarder $> --- <$ award.title $>} \hfill {\em <$ printDate(award.date) $>} \\ 
<$ award.summary $>

<% endfor %>


\end{rSection}

%----------------------------------------------------------------------------------------

%----------------------------------------------------------------------------------------
%	SKILLS SECTION
%----------------------------------------------------------------------------------------

\begin{rSection}{Skills}

\begin{tabular}{ @{} >{\bfseries}l @{\hspace{3ex}} l }
<% set filteredSkills = skills | ConvertSkills %>
<% for skill  in filteredSkills %>
<$ skill.name $> & \\
<% for entry in skill.entries %>
<% if loop.last %> \smallbreak <% endif %>
\textnormal{  <$ entry.level $>} &
<%- set comma = joiner() -%> 
<%- for keyw in entry.keywords -%> 
<$ comma() $> <$ keyw | TeXEscape $> 
<%- endfor -%>\\
<%- endfor %>
<% endfor %>
\end{tabular}

\end{rSection}

%----------------------------------------------------------------------------------------
%	EDUCATION SECTION
%----------------------------------------------------------------------------------------

\begin{rSection}{Education}

<% for school in education %>

{\bf <$ school.institution $>} \hfill {\em <$ printDate(school.startDate) $> --- <$ printDate(school.endDate) $>} \\ 
<$ school.studyType $> in <$ school.area $> \hfill Overall GPA: <$ school.gpa $>

<% endfor %>

\end{rSection}

%----------------------------------------------------------------------------------------
%	LANGUAGES SECTION
%----------------------------------------------------------------------------------------

\begin{rSection}{Languages}

\begin{tabular}{ @{} >{\bfseries}l @{\hspace{6ex}} l }
<% for lang  in languages %>
<$ lang.language $> & <$ lang.fluency $> \\
<% endfor %>
\end{tabular}


\end{rSection}


%----------------------------------------------------------------------------------------
%	INTERESTS SECTION
%----------------------------------------------------------------------------------------

\begin{rSection}{Interests}

\begin{tabular}{ @{} >{\bfseries}l @{\hspace{6ex}} l }
<% for interest  in interests %>
<$ interest.name $> & 
<%- set comma = joiner() -%> 
<%- for keyw in interest.keywords -%> 
<$ comma() $> <$ keyw $> 
<%- endfor -%>\\
<% endfor %>
\end{tabular}

\end{rSection}

%----------------------------------------------------------------------------------------
%	PROFILES SECTION
%----------------------------------------------------------------------------------------

\begin{rSection}{Profiles}

\begin{tabular}{ @{} >{\bfseries}l @{\hspace{6ex}} l }
<% for profile  in basics.profiles -%>
<$  profile.network $> & <$ profile.url $>\\
\textnormal{  <$ profile.username $>} & \\
<%- endfor %>
\end{tabular}

\end{rSection}
